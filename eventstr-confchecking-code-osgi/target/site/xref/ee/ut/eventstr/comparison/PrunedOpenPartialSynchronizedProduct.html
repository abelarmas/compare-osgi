366 		State newState = new State(c1, labels, c2);
367 		states.add(newState);
368 
369 		newState.action = StateHint.CREATED;
370 		
371 		if (stateSpaceTable.contains(c1, c2)) {
372 			Map<Multiset<String>, State> map = stateSpaceTable.get(c1, c2);
373 			if (map.containsKey(labels))
374 				newState.action = StateHint.MERGED;
375 			else
376 				map.put(labels, newState);
377 		} else {
378 			Map<Multiset<String>, State> map = new HashMap<>();
379 			map.put(labels, newState);
380 			stateSpaceTable.put(c1, c2, map);
381 		}
382 		return newState;
383 	}

425 		Set<State> gvisited = new HashSet<>();
426 		Stack<State> open = new Stack<>();
427 		this.opSeq = new LinkedList<>();
428 
429 		for (int i = 0; i < numberOfTargets; i++) {
430 			State s = matchings;
431 			if (s == null) continue;
432 			open.push(s);
433 			Set<State> visited = new HashSet<>();
434 			while (!open.isEmpty()) {
435 				State curr = open.pop();
436 				visited.add(curr);
437 				
438 				for (State pred: ancestors.get(curr)) {
439 					if (!visited.contains(pred) && !open.contains(pred)) {
440 						open.push(pred);
441 						for (Operation op: descendants.get(pred)) {
442 							if (op.nextState.equals(curr)) {
443 								opSeq.addFirst(op);
444 								break;
445 							}
446 						}
447 					}
448 				}
449 			}
450 			
451 
452 			gvisited.addAll(visited);
453 		}
454 		
455 		this.relevantStates = gvisited;
456
