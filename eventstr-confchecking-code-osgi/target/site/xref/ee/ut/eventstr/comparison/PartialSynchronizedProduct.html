111 	public PartialSynchronizedProduct<T> perform() {
112 		Map<BitSet, Float> lGW = new HashMap<>();
113 		Map<BitSet, Float> rGW = new HashMap<>();
114 
115 		for (BitSet mconf: pes1.getMaxConf())
116 			lGW.put(mconf, Float.POSITIVE_INFINITY);
117 		for (BitSet mconf: pes2.getMaxConf())
118 			rGW.put(mconf, Float.POSITIVE_INFINITY);
119 
120 		Queue<State> open = new PriorityQueue<State>(11, new WeightBasedStateComparator());
121 
122 		root = getState(new BitSet(), HashMultiset.<String> create(), new BitSet()).getSecond();
123 		open.offer(root);
124 
125 		while (!open.isEmpty()) {
126 			State s = open.poll();
127 
128 			if (isCandidate(s.c1, s, pes1, lGW) || isCandidate(s.c2, s, pes2, rGW)) {
129 				if (pes1.getMaxConf().contains(s.c1) && pes2.getMaxConf().contains(s.c2)) {
130 //					updateMatches(s.c1, s, lMATCHES, lGW);
131 //					updateMatches(s.c2, s, rMATCHES, rGW);
132 					updateMatches(s, lGW, rGW);
133 				}
134 				
135 				if (s.hint == StateHint.MERGED) continue;
136 				
137 				BitSet lpe = pes1.getPossibleExtensions(s.c1);
138 				BitSet rpe = pes2.getPossibleExtensions(s.c2);
139 				
140 				for (int e1 = lpe.nextSetBit(0); e1 >= 0; e1 = lpe.nextSetBit(e1+1)) {
141 					String label1 = pes1.getLabel(e1);
142 					BitSet c1p = (BitSet)s.c1.clone();
143 					c1p.set(e1);
144 					
145 					/// HIDE (left)
146 					Pair<StateHint, State> pair = getState(c1p, s.labels, s.c2);
147 					switch (pair.getFirst()) {
148 					case CREATED:
149 					case MERGED:
150 						open.offer(pair.getSecond()); // Only if not previously visited
151 						descendants.put(s, Operation.lhide(pair.getSecond(), e1, label1));
152 						ancestors.put(pair.getSecond(), s);
153 					default:
154 					}
155 					
156 					for (int e2 = rpe.nextSetBit(0); e2 >= 0; e2 = rpe.nextSetBit(e2+1)) {
157 						if (label1.equals(pes2.getLabel(e2)) 
158 								&& isOrderPreserving(s, e1, e2)) {
159 							/// MATCH !!!
160 							BitSet c2p = (BitSet)s.c2.clone();
161 							c2p.set(e2);
162 
163 							Multiset<String> labels = HashMultiset.create(s.labels);
164 							labels.add(label1);
165 							
166 							pair = getState(c1p, labels, c2p);
167 							
168 							switch (pair.getFirst()) {
169 							case CREATED:
170 							case MERGED:
171 								open.offer(pair.getSecond()); // Only if not previously visited
172 								descendants.put(s, Operation.match(pair.getSecond(), new Pair<>(e1, e2), label1));
173 								ancestors.put(pair.getSecond(), s);
174 							default:
175 							}
176 						}
177 					}
178 				}
179 				for (int e2 = rpe.nextSetBit(0); e2 >= 0; e2 = rpe.nextSetBit(e2+1)) {
180 					/// HIDE (right)
181 					BitSet c2p = (BitSet)s.c2.clone();
182 					c2p.set(e2);
183 
184 					Pair<StateHint, State> pair = getState(s.c1, s.labels, c2p);
185 					switch (pair.getFirst()) {
186 					case CREATED:
187 					case MERGED:
188 						open.offer(pair.getSecond()); // Only if not previously visited
189 						descendants.put(s, Operation.rhide(pair.getSecond(), e2, pes2.getLabel(e2)));
190 						ancestors.put(pair.getSecond(), s);
191 					default:
192 					}
193 				}
194 			}
195 //			IOUtils.toFile("psp1.dot", toDot());
196 //			System.out.print(".");
197 			// else ... the State is discarded
198 		}

203 	private boolean isOrderPreserving(State s, int e1, int e2) {
204 		Stack<State> open = new Stack<>();
205 		Set<State> visited = new HashSet<>();
206 		open.push(s);
207 		while (!open.isEmpty()) {
208 			State curr = open.pop();
209 			visited.add(curr);
210 			
211 			for (State ancestor: ancestors.get(curr)) {
212 				if (visited.contains(ancestor) || open.contains(ancestor)) continue;
213 				for (Operation op: descendants.get(ancestor))
214 					if (op.op == Op.MATCH && op.nextState.equals(curr)) {
215 						@SuppressWarnings("unchecked")
216 						Pair<Integer, Integer> matchedEvents = (Pair<Integer,Integer>)op.target;
217 						
218 						if (!pes1.getBRelation(e1, matchedEvents.getFirst()).equals(pes2.getBRelation(e2, matchedEvents.getSecond())))
219 							return false;
220 					}
221 				open.push(ancestor);
222 			}
223 		}
224 		return true;
225 	}

228 		Set<State> gvisited = new HashSet<>();
229 		Set<State> finalStates = new HashSet<>(lMATCHES.values());
230 		finalStates.addAll(rMATCHES.values());
231 		Stack<State> open = new Stack<>();
232 		for (State s: finalStates) {
233 			open.push(s);
234 			Set<State> visited = new HashSet<>();
235 			while (!open.isEmpty()) {
236 				State curr = open.pop();
237 				visited.add(curr);
238 				
239 				if (mergedStates.get(curr).size() > 1) {
240 					for (State merged: mergedStates.get(curr))
241 						if (!curr.equals(merged)) open.push(merged);
242 				}
243 
244 				for (State pred: ancestors.get(curr))
245 					if (!visited.contains(pred) && !open.contains(pred))
246 						open.push(pred);
247 			}
248 			
249 
250 			gvisited.addAll(visited);
251 		}
252 		
253 		this.relevantStates = gvisited;
254 		
255 		System.out.println("Number of final states: " + rMATCHES.values().size());
256 		System.out.println("Number of relevant states: " + relevantStates.size());

261 	private Pair<StateHint,State> getState(BitSet c1, Multiset<String> labels, BitSet c2) {
262 		StateHint action = StateHint.CREATED;
263 		State newState = eta(new State(c1, labels, c2));
264 		states.add(newState);
265 		newState.hint = StateHint.CREATED;
266 		
267 		if (stateSpaceTable.contains(c1, c2)) {
268 			State pivot = stateSpaceTable.get(c1, c2);
269 			
270 			if (newState.weight == pivot.weight) {
271 				boolean found = false;
272 				for (State sibling: mergedStates.get(pivot))
273 					if (sibling.labels.containsAll(labels)) {
274 						found = true;
275 						break;
276 					}
277 				if (!found) {
278 					newState.hint = action = StateHint.MERGED;
279 					mergedStates.put(pivot, newState);
280 				} else
281 					newState.hint = action = StateHint.DISCARDED;				
282 			} else {
283 				newState.hint = action = StateHint.DISCARDED;
284 			}
285 		} else {
286 			stateSpaceTable.put(c1, c2, newState);
287 			mergedStates.put(newState, newState);
288 		}
289 		return new Pair<>(action, newState);
290 	}

331 		Set<String> pf1 = pes1.getPossibleFutureAsLabels(s.c1);
332 		Set<String> pf2 = pes2.getPossibleFutureAsLabels(s.c2);

337 	public float g(BitSet c1, BitSet c2, Multiset<String> labels) {
338 		return (c1.cardinality() + c2.cardinality() - labels.size() * 2.0f) / gDenominator;
339 	}

341 	public float h(Set<String> pf1, Set<String> pf2) {
342 		Set<String> union = pf1;
343 		Set<String> intersection = new HashSet<>(pf1);
344 		intersection.retainAll(pf2);
345 		union.addAll(pf2);
346 		return (union.size() - intersection.size() + 0.0f) / gDenominator;
347 	}

349 	public float hAbel(Set<String> pf1, Set<String> pf2) {
350 		Set<String> union = pf1;
351 		Set<String> intersection = new HashSet<>(pf1);
352 		intersection.retainAll(pf2);
353 		union.addAll(pf2);
354 		return 1.0f - ((intersection.size() + 0.0f)/union.size());
355 	}

364 	private void updateMatches(State s, Map<BitSet, Float> lGW, Map<BitSet, Float> rGW) {
365 		if (s.weight < lGW.get(s.c1) || s.weight < rGW.get(s.c2)) {
366 			if (s.weight < lGW.get(s.c1)) {
367 				lMATCHES.removeAll(s.c1);
368 				lGW.put(s.c1, s.weight);
369 			}
370 			if (s.weight < rGW.get(s.c2)) {
371 				rMATCHES.removeAll(s.c2);
372 				rGW.put(s.c2, s.weight);
373 			}
374 			lMATCHES.put(s.c1, s);
375 			rMATCHES.put(s.c2, s);
376 		} else {
377 			if (s.weight == lGW.get(s.c1))
378 				lMATCHES.put(s.c1, s);
379 			if (s.weight == rGW.get(s.c2))
380 				rMATCHES.put(s.c2, s);
381 		}
382 	}

402 		int cardinality = superset.cardinality();
403 		superset = (BitSet)superset.clone();
404 		superset.or(subset);
405 		return cardinality == superset.cardinality();

420 		public MaximalStateAnalyzer partitionStateSpace(Collection<State> maxStates) {
421 			Stack<State> open = new Stack<>();
422 			for (State mstate: maxStates)
423 				open.push(mstate);
424 			
425 			while (!open.isEmpty()) {
426 				State curr = open.pop();
427 				visited.add(curr);
428 				if (mergedStates.get(curr).size() > 1) {
429 					for (State merged: mergedStates.get(curr))
430 						if (!curr.equals(merged)) open.push(merged);
431 				}
432 				for (State pred: ancestors.get(curr))
433 					if (!visited.contains(pred) && !open.contains(pred))
434 						open.push(pred);
435 			}			
436 			return this;
437 		}

439 		public MaximalStateAnalyzer forwardAnalysis() {
440 			Queue<State> open = new LinkedList<>();
441 			open.offer(root);
442 			while (!open.isEmpty()) {
443 				State curr = open.poll();
444 				
445 				// Look for earliest mismatch
446 				if (earliestMissmatch == null)
447 					for (Operation op: descendants.get(curr))
448 						if (visited.contains(op.nextState) && op.op != Op.MATCH) {
449 							earliestMissmatch = op;
450 							break;
451 						}
452 				
453 				// Continue the traversal and gather list of matchings
454 				for (Operation op: descendants.get(curr))
455 					if (visited.contains(op.nextState)) {
456 						if (op.op == Op.MATCH) {
457 							@SuppressWarnings("unchecked")
458 							Pair<Integer, Integer> pair = (Pair<Integer,Integer>)op.target;
459 							localMatches.add(pair);
460 						}
461 						open.offer(op.nextState);
462 						labelOperationMap.put(op.label, op);
463 					}
464 			}
465 			return this;
466 		}

478 			return this;
