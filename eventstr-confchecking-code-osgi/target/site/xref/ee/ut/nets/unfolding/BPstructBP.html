60  	/**
61  	 * The search strategy for
62  	 * {@link #equivalentCuts_conditionSignature_history(byte[], DNode[], DNode[])}
63  	 * . A size-based search strategy ({@link Options#searchStrat_size}).
64  	 * 
65  	 * The method has been extended to determine cut-off events by a
66  	 * lexicographic search strategy.
67  	 * 
68  	 * @param newEvent
69  	 *            event that has been added to the branching process
70  	 * @param newCut
71  	 *            the cut reached by 'newEvent'
72  	 * @param eventsToCompare
73  	 * @return <code>true</code> iff <code>newEvent</code> is a cut-off event
74  	 *         because of some equivalent event in <code>eventsToCompare</code>.
75  	 */
76  	protected boolean findEquivalentCut_bpstruct(int newCutConfigSize,
77  			DNode newEvent, DNode[] newCut, Iterable<DNode> eventsToCompare) {
78  		// all extensions to support the size-lexicographic search-strategy
79  		// or marked with LEXIK
80  
81  		// optimization: determine equivalence of reached cuts by the
82  		// help of a 'condition signature, initialize 'empty' signature
83  		// for 'newEvent', see #equivalentCuts_conditionSignature_history
84  
85  		byte[] newCutSignature = cutSignature_conditions_init255();
86  
87  		// compare the cut reached by 'newEvent' to the initial cut
88  		if (newCut.length == bp.initialCut.length)
89  			if (equivalentCuts_conditionSignature_history(newCutSignature,
90  					newCut, bp.initialCut)) {
91  				// yes, newEvent reaches the initial cut again
92  				setCutOffConditions(newEvent, newCut, bp.initialCut);
93  				return true; // 'newEvent' is a cut-off event
94  			}
95  
96  		// Check whether 'eventsToCompare' contains a "smaller" event that
97  		// reaches
98  		// the same cut as 'newEvent'.
99  
100 		// Optimization: to quickly avoid comparing configurations of events
101 		// that
102 		// do not reach the same cut as 'newEvent', compare and store hash
103 		// values
104 		// of the reached configurations.
105 
106 		// get hash value of the cut reached by 'newEvent'
107 		int newEventHash = getPrimeConfiguration_CutHash().get(newEvent);
108 
109 		// check whether 'newEvent' is a cut-off event by comparing to all other
110 		// given events
111 		Iterator<DNode> it = eventsToCompare.iterator();
112 
113 		while (it.hasNext()) {
114 			DNode e = it.next();
115 
116 			// do not check the event that has just been added, the cuts would
117 			// be equal...
118 			if (e == newEvent)
119 				continue;
120 
121 			// newCut is only equivalent to oldCut if the configuration of
122 			// newCut
123 			// is (lexicographically) larger than the configuration of oldCut
124 			if (!getPrimeConfiguration_Size().containsKey(e)) {
125 				// the old event 'e' has incomplete information about its prime
126 				// configuration, cannot be used to check for cutoff
127 				continue;
128 			}
129 			// optimization: compared hashed values of the sizes of the prime
130 			// configurations
131 			if (newCutConfigSize < getPrimeConfiguration_Size().get(e)) {
132 				// the old one is larger, not equivalent
133 				continue;
134 			}
135 
136 			// optimization: compare reached states by their hash values
137 			// only if hash values are equal, 'newEvent' and 'e' could be
138 			// equivalent
139 			if (getPrimeConfiguration_CutHash().get(e) != newEventHash)
140 				continue;
141 
142 			// retrieve the cut reached by the old event 'e'
143 			DNode[] oldCut = bp.getPrimeCut(e,
144 					getOptions().searchStrat_lexicographic,
145 					getOptions().searchStrat_lexicographic);
146 
147 			// cuts of different lengths cannot reach the same state, skip
148 			if (newCut.length != oldCut.length)
149 				continue;
150 
151 			// if both configurations have the same size:
152 			if (newCutConfigSize == bp.getPrimeConfiguration_size) {
153 				// and if not lexicographic, then the new event cannot be
154 				// cut-off event
155 				if (!getOptions().searchStrat_lexicographic)
156 					continue;
157 				// LEXIK: otherwise compare whether the old event's
158 				// configuration is
159 				// lexicographically smaller than the new event's configuration
160 				if (!isSmaller_lexicographic(
161 						getPrimeConfigurationString().get(e),
162 						getPrimeConfigurationString().get(newEvent))) {
163 					// Check whether 'e' was just added. If this is the case,
164 					// then 'e' and 'newEvent'
165 					// were added in the wrong order and we check again whether
166 					// 'e' is a cut-off event
167 					if (e.post != null && e.post.length > 0 && e.post[0]._isNew) {
168 						// System.out.println("smaller event added later, should switch cut-off");
169 						checkForCutOff_again(e);
170 					}
171 					continue;
172 				}
173 			}
174 
175 			boolean doRestrict = true;
176 
177 			// if (cyclicPhase && (cyclicNodes.contains(properName(newEvent)) ||
178 			// cyclicNodes.contains(properName(e)))) {
179 			if (mode == MODE.ONEUNFOLDING) {
180 				if (doRestrict = isCorrInLocalConfig(newEvent, e))
181 					doRestrict = checkContainment(newEvent, e, newCut, oldCut);
182 			} else if (mode == MODE.ZEROUNFOLDING)
183 				doRestrict = isCorrInLocalConfig(newEvent, e);
184 			else if (mode == MODE.HALFUNFOLDING)
185 				doRestrict = checkContainment(newEvent, e, newCut, oldCut);
186 
187 			//
188 			// cyclicNodes.add(properName(newEvent)); //
189 			// } else
190 			// doRestrict = checkAcyclicCase(newEvent,e,newCut,oldCut);
191 
192 			// The prime configuration of 'e' is either smaller or
193 			// lexicographically
194 			// smaller than the prime configuration of 'newEvent'. Further, both
195 			// events
196 			// reach cuts of the same size. Check whether both cuts reach the
197 			// same histories
198 			// by comparing their condition signatures
199 			if (doRestrict
200 					&& equivalentCuts_conditionSignature_history(
201 							newCutSignature, newCut, oldCut)) {
202 				// yes, equivalent cuts, make events and conditions equivalent
203 				setCutOffEvent(newEvent, e);
204 				setCutOffConditions(newEvent, newCut, oldCut);
205 				// and yes, 'newEvent' is a cut-off event
206 				return true;
207 			}
208 		}
209 
210 		// no smaller equivalent has been found
211 		return false;
212 	}

243 	/**
244 	 * Restrict cutoff criterion for cyclic case
245 	 * 
246 	 * A cutoff is cyclic if either cutoff or its corresponding event refer to a
247 	 * transition of the originative net that is part of some cyclic path of the
248 	 * net
249 	 * 
250 	 * @param cutoff
251 	 *            Cutoff event
252 	 * @param corr
253 	 *            Corresponding event
254 	 * @param cutoff_cut
255 	 *            Cutoff cut
256 	 * @param corr_cut
257 	 *            Corresponding cut
258 	 * @return <code>true</code> if cyclic cutoff criterion holds; otherwise
259 	 *         <code>false</code>
260 	 */
261 	protected boolean checkContainment(DNode cutoff, DNode corr,
262 			DNode[] cutoff_cut, DNode[] corr_cut) {
263 
264 		Set<DNode> lconf_cutoff = getLocalConfig(cutoff);
265 		Set<DNode> lconf_corr = getLocalConfig(corr);
266 
267 		Set<Integer> transSet_cutoff = new HashSet<Integer>();
268 		for (DNode ev : lconf_cutoff)
269 			transSet_cutoff.add((int) ev.id);
270 
271 		Set<Integer> transSet_corr = new HashSet<Integer>();
272 		for (DNode ev : lconf_corr)
273 			transSet_corr.add((int) ev.id);
274 
275 		return transSet_corr.containsAll(transSet_cutoff);
276 	}

278 	public Set<DNode> getLocalConfig(DNode event) {
279 		Stack<DNode> stack = new Stack<DNode>();
280 		Set<DNode> visited = new HashSet<DNode>();
281 		Set<DNode> events = new HashSet<DNode>();
282 		stack.push(event);
283 		while (!stack.isEmpty()) {
284 			DNode curr = stack.pop();
285 			visited.add(curr);
286 			if (curr.isEvent)
287 				events.add(curr);
288 			if (curr.pre == null)
289 				continue;
290 			for (DNode p : curr.pre)
291 				if (!visited.contains(p) && !stack.contains(p))
292 					stack.push(p);
293 		}
294 		return events;
295 	}

298 	 * Check whether cutoff or its corresponding event refer to a gateway node
299 	 * in the originative net; false otherwise
300 	 * 
301 	 * @param cutoff
302 	 *            Cutoff event
303 	 * @param corr
304 	 *            Corresponding event
305 	 * @return <code>true</code> if cutoff or its corresponding event refer to a
306 	 *         gateway node; otherwise <code>false</code>
307 	 */
308 	protected boolean checkGateway(DNode cutoff, DNode corr) {
309 		if (cutoff.post.length > 1 || cutoff.pre.length > 1
310 				|| corr.post.length > 1 || corr.pre.length > 1)
311 			return true;
312 
313 		return false;
314 	}

316 	/**
317 	 * Check if conditions in cuts of cutoff and corresponding events are
318 	 * shared, except of postsets
319 	 * 
320 	 * @param cutoff
321 	 *            Cutoff event
322 	 * @param corr
323 	 *            Corresponding event
324 	 * @param cutoff_cut
325 	 *            Cutoff cut
326 	 * @param corr_cut
327 	 *            Corresponding cut
328 	 * @return <code>true</code> if shared; otherwise <code>false</code>
329 	 */
330 	protected boolean checkConcurrency(DNode cutoff, DNode corr,
331 			DNode[] cutoff_cut, DNode[] corr_cut) {
332 		Set<Integer> cutoffSet = new HashSet<Integer>();
333 		Set<Integer> corrSet = new HashSet<Integer>();
334 
335 		int i = 0;
336 		for (i = 0; i < cutoff_cut.length; i++)
337 			cutoffSet.add(cutoff_cut[i].globalId);
338 		for (i = 0; i < corr_cut.length; i++)
339 			corrSet.add(corr_cut[i].globalId);
340 		for (i = 0; i < cutoff.post.length; i++)
341 			cutoffSet.remove(cutoff.post[i].globalId);
342 		for (i = 0; i < corr.post.length; i++)
343 			corrSet.remove(corr.post[i].globalId);
344 
345 		if (cutoffSet.size() != corrSet.size())
346 			return false;
347 		for (Integer n : cutoffSet) {
348 			if (!corrSet.contains(n))
349 				return false;
350 		}
351 
352 		return true;
353 	}

355 	/**
356 	 * Check if corresponding event is in the local configuration of the cutoff
357 	 * 
358 	 * @param cutoff
359 	 *            Cutoff event
360 	 * @param corr
361 	 *            Corresponding event
362 	 * @return <code>true</code> if corresponding event is in the local
363 	 *         configuration of the cutoff; otherwise <code>false</code>
364 	 */
365 	protected boolean isCorrInLocalConfig(DNode cutoff, DNode corr) {
366 		List<Integer> todo = new ArrayList<Integer>();
367 		Map<Integer, DNode> i2d = new HashMap<Integer, DNode>();
368 		for (DNode n : Arrays.asList(cutoff.pre)) {
369 			todo.add(n.globalId);
370 			i2d.put(n.globalId, n);
371 		}
372 		Set<Integer> visited = new HashSet<Integer>();
373 
374 		while (!todo.isEmpty()) {
375 			Integer n = todo.remove(0);
376 			visited.add(n);
377 
378 			if (n.equals(corr.globalId))
379 				return true;
380 
381 			for (DNode m : i2d.get(n).pre) {
382 				if (!visited.contains(m.globalId)) {
383 					todo.add(m.globalId);
384 					i2d.put(m.globalId, m);
385 				}
386 			}
387 		}
388 
389 		return false;
390 	}

412 		StringBuilder b = new StringBuilder();
413 		b.append("digraph BP {\n");
414 
415 		// standard style for nodes and edges
416 		b.append("graph [fontname=\"Helvetica\" nodesep=0.3 ranksep=\"0.2 equally\" fontsize=10];\n");
417 		b.append("node [fontname=\"Helvetica\" fontsize=8 fixedsize width=\".3\" height=\".3\" label=\"\" style=filled fillcolor=white];\n");
418 		b.append("edge [fontname=\"Helvetica\" fontsize=8 color=white arrowhead=none weight=\"20.0\"];\n");

421 		// "fillcolor=black peripheries=2 height=\".2\" width=\".2\" ";
422 		String cutOffFillString = "fillcolor=gold";
423 		String antiFillString = "fillcolor=red";
424 		String impliedFillString = "fillcolor=violet";
425 		String hiddenFillString = "fillcolor=grey";
426 
427 		// first print all conditions
428 		b.append("\n\n");
429 		b.append("node [shape=circle];\n");
430 		for (DNode n : bp.allConditions) {
431 			if (!option_printAnti && n.isAnti)
432 				continue;
433 			/*
434 			 * - print current marking if (cutNodes.contains(n))

439 			else if (n.isCutOff)
440 				b.append("  c" + n.globalId + " [" + cutOffFillString + "]\n");
441 			else
442 				b.append("  c" + n.globalId + " []\n");
443 
444 //			String auxLabel = "";
445 
446 			b.append("  c" + n.globalId + "_l [shape=none];\n");
447 			// Diagrams
448 			// b.append("  c"+n.globalId+"_l -> c"+n.globalId+" [headlabel=\""+n+" "+auxLabel+"\"]\n");
449 		}
450 
451 		// then print all events
452 		b.append("\n\n");
453 		b.append("node [shape=box];\n");
454 		for (DNode n : bp.allEvents) {
455 			if (!option_printAnti && n.isAnti)
456 				continue;
457 			if (n.isAnti && n.isHot)
458 				b.append("  e" + n.globalId + " [" + antiFillString + "]\n");

460 				b.append("  e" + n.globalId + " [" + hiddenFillString + "]\n");

471 			// Diagrams
472 			b.append("  e"+n.globalId+"_l -> e"+n.globalId+" [headlabel=\""+n+" "+auxLabel+"\"]\n");
473 //			b.append(" label=\""+n+"\"];\n");
474 		}
475

489 							+ n.globalId + " [weight=10000.0]\n");
490 				else
491 					b.append("  c" + n.pre[i].globalId + " -> " + prefix
492 							+ n.globalId + " [weight=10000.0]\n");
493 			}
494 		}
495 
496 		for (DNode n : bp.allEvents) {
497 			String prefix = n.isEvent ? "e" : "c";
498 			for (int i = 0; i < n.pre.length; i++) {
499 				if (n.pre[i] == null)
500 					continue;
501 				if (!option_printAnti && n.isAnti)
502 					continue;
503 				if (n.pre[i].isEvent)
504 					b.append("  e" + n.pre[i].globalId + " -> " + prefix
505 							+ n.globalId + " [weight=10000.0]\n");
506 				else
507 					b.append("  c" + n.pre[i].globalId + " -> " + prefix
508 							+ n.globalId + " [weight=10000.0]\n");
509 			}

511

513 		b.append("\n\n");
514 		b.append(" edge [fontname=\"Helvetica\" fontsize=8 arrowhead=normal color=red];\n");
515 		for (DNode n : bp.allEvents) {
516 			if (n.isCutOff
517 					&& futureEquivalence().getElementary_ccPair().get(n) != null) {
518 				if (!this.isCorrInLocalConfig(n, futureEquivalence()
519 						.getElementary_ccPair().get(n)))
520 					b.append("  e"
521 							+ n.globalId
522 							+ " -> e"
523 							+ futureEquivalence().getElementary_ccPair().get(n).globalId
524 							+ " [weight=10000.0]\n");
525 			}
526 		}
527 
528 		// and add links from cutoffs to corresponding events (causal case)
529 		b.append("\n\n");
530 		b.append(" edge [fontname=\"Helvetica\" fontsize=8 arrowhead=normal color=blue];\n");
531 		for (DNode n : bp.allEvents) {
532 			if (n.isCutOff
533 					&& futureEquivalence().getElementary_ccPair().get(n) != null) {
534 				if (this.isCorrInLocalConfig(n, futureEquivalence()
535 						.getElementary_ccPair().get(n)))
536 					b.append("  e"
537 							+ n.globalId
538 							+ " -> e"
539 							+ futureEquivalence().getElementary_ccPair().get(n).globalId
540 							+ " [weight=10000.0]\n");
541 			}
542 		}
543 
544 		b.append("}");
545 		return b.toString();
546 	}
